177 # @return Sorted list, ascending
178 #
179 def sorted(l0):
180  l = []
181  for x in l0:
182  l += [x]
183  nl = []
184  while len(l) > 0:
185  mi=0
186  m=l[0]
187  for i in range(1, len(l)):
188  if l[i] < m:
189  m = l[i]

191  l = l[0:mi] + l[mi+1:len(l)]
192  nl = nl + [m]
