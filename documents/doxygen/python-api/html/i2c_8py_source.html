1 ## @file
2 # @package i2c
3 #
4 # @brief Generic I2C bus driver using /dev and ioctl.
5 #
6 # General:
7 #
8 # Broadly speaking this driver follows the precedents set by the internal
9 # Linux API for device drivers that access their devices via I2C.
10 #
11 # For maximum flexibility, this driver does not encode the paths to the
12 # i2c nodes. This is because it will often be desirable to make such
13 # paths a config option, or to use a symbolic link or other alias in
14 # lieu of the usual path.
15 #
16 # Note: this driver does not keep track of I2C addresses. This is a more
17 # natural fit for individual device drivers. Firstly, the I2C base address
18 # is hard-coded into i2c ICs. Secondly, the address offset, if any, is
19 # managed by the IC according to its own documentation (eg as a function
20 # of Ax pins). Finally, while the Linux i2c driver model carries a device
21 # address in its equivalent of our "device" class, that address is *overridden*
22 # by the one passed to i2c_transfer() in the i2c_msg struct. We are in fact
23 # consistent in behaviour because of this.
24 #
25 # Python:
26 #
27 # The i2c_msg structure has been encapsuated in free function msg() and
28 # the i2c_rdwr_ioctl_data structure has been encapsulated in device.__init__().
29 # This avoids exposing the ioctl block-building process.
30 #
31 # Copyright (c) Open IO Labs Ltd 2016
32 # All Rights Reserved
33 #

90  ## @brief Perform an i2c transfer
91  #
92  # @param msgs A list of message objects
93  #
94  # The message objects must have been created using the msg() function.
95  # A list is always required, even if only one msg is present.
96  # This function corresponds to the i2c_transfer() function defined in
97  # the Linux kernel. The i2c_rdwr_ioctl_data structure is created
98  # automatically from the supplied message list.
99  def transfer( self, msgs ):
100  # Create struct i2c_rdwr_ioctl_data
101  rdwr_arg = [devsysfs.block(msgs),uint32(len(msgs))]
102  self.dev.ioctl( I2C_RDWR, devsysfs.block(rdwr_arg) )

104  ## @brief Create a single I2C message
105  #
106  # @param addr I2c address for transfer (not including R/W bit)
107  # @param flags Any OR-combinations of the I2C_M_ flags
108  # @param buf A list of uint8 types serving as buffer for the transfer
109  #
110  # The size of the buffer is the length of the transfer, and so a buffer
111  # must be provided for reads as well as writes. The buffer is mutated
112  # by a read transfer, regardless of any apparent immutability of its
113  # elements. The message created by this call corresponds to the i2c_msg
114  # structure defined in the Linux kernel.
115  def msg( self, addr, flags, buf ):
116  # Create strcut i2c_msg
117  return [uint16(addr), uint16(flags), uint16(len(buf)), devsysfs.block(buf)]

189  # ====================
190  #
191  # This reads a single byte from a device, without specifying a device
192  # register. Some devices are so simple that this interface is enough; for
193  # others, it is a shorthand if you want to read the same register as in
194  # the previous SMBus command.
195  #
196  # Bus sequence `S Addr Rd [A] [Data] NA P`

202  self.smbus_access(I2C_SMBUS_READ, 0, I2C_SMBUS_BYTE, [byte])

302  ## SMBus Block Read:
303  # ==================
304  #
305  # This command reads a block of up to 32 bytes from a device, from a
306  # designated register that is specified through the Comm byte. The amount
307  # of data is specified by the device in the Count byte.
308  #
309  # Bus sequence: `S Addr Wr [A] Comm [A] S Addr Rd [A] [Count] A [Data] A [Data] A ... A [Data] NA P`
310  #
311  # @param command The register to read from
312  # @return The number of read bytes, and a list containing the read bytes
313  #
314  def smbus_read_block_data(self, command):
315  data = []
316  for i in range(I2C_SMBUS_BLOCK_MAX_PADDED):
317  data = data + [uint8()]
318  self.smbus_access(I2C_SMBUS_READ, command, I2C_SMBUS_BLOCK_DATA, data)
319  return data[0], data[1:]
