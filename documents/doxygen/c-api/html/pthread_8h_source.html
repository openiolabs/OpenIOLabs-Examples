554 int pthread_cond_wait(pthread_cond_t *cond,
555  pthread_mutex_t *mutex)
556 {
557  // release the mutex
558  *mutex = false;
559  // Wait for a signal
560  cond->waiters++;
561  sml_coroutine_wait( cond->triggers != 0 );
562  // consume the signal
563  cond->triggers--;
564  cond->waiters--;
565  // wait for the mutex and re-lock
566  sml_coroutine_wait( !(*mutex) );
567  *mutex = true;
568  return 0;
569 }
